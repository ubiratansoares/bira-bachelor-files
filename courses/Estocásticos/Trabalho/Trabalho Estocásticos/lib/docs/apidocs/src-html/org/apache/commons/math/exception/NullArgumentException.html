017    package org.apache.commons.math.exception;
018    
019    import org.apache.commons.math.exception.util.Localizable;
020    import org.apache.commons.math.exception.util.LocalizedFormats;
021    
022    /**
023     * All conditions checks that fail due to a {@code null} argument must throw
024     * this exception.
025     * This class is meant to signal a precondition violation ("null is an illegal
026     * argument") and so does not extend the standard {@code NullPointerException}.
027     * Proagation of {@code NullPointerException} from within Commons-Math is
028     * construed to be a bug.
029     *
030     * @since 2.2
031     * @version $Revision$ $Date$
032     */
033    public class NullArgumentException extends MathIllegalArgumentException {
034        /** Serializable version Id. */
035        private static final long serialVersionUID = -6024911025449780478L;
036    
037        /**
038         * Default constructor.
039         */
040        public NullArgumentException() {
041            super(LocalizedFormats.NULL_NOT_ALLOWED);
042        }
043        /**
044         * @param specific Message pattern providing the specific context of
045         * the error.
046         */
047        public NullArgumentException(Localizable specific) {
048            super(specific, LocalizedFormats.NULL_NOT_ALLOWED);
049        }
050    }
